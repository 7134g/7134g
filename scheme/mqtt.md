#### 分布式 emqttd
    yum -y install make gcc gcc-c++ kernel-devel m4 ncurses-devel openssl-devel
    wget http://www.erlang.org/download/otp_src_R13B04.tar.gz
    tar xfvz otp_src_R13B04.tar.gz
    cd otp_src_R13B04/
    ./configure --with-ssl
    sudo make install
